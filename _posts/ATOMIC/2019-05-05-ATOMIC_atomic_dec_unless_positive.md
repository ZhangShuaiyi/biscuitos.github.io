# <span id="源码分析">源码分析</span>

{% highlight ruby %}
#ifndef atomic_dec_unless_positive
static inline bool atomic_dec_unless_positive(atomic_t *v)
{
        int c = atomic_read(v);

        do {
                if (unlikely(c > 0))
                        return false;
        } while (!atomic_try_cmpxchg(v, &c, c - 1));

        return true;
}
#endif
{% endhighlight %}

atomic_dec_unless_positive() 函数用于对负数执行减一操作。函数首先判断 atomic_t
变量的值是否为正数，如果是，则直接返回；如果不是，则调用 atomic_try_cmpxchg()
函数进行减一操作。

#include <linux/kernel.h>
#include <linux/init.h>

static atomic_t BiscuitOS_counter = ATOMIC_INIT(2);

/* atomic_* */
static __init int atomic_demo_init(void)
{
	/* Legacy data */
	printk("[0]Atomic: %d\n", atomic_read(&BiscuitOS_counter));

	atomic_dec_unless_positive(&BiscuitOS_counter);

	/* New data */
	printk("[1]Atomic: %d\n", atomic_read(&BiscuitOS_counter));

	/* add > 0 */
	atomic_sub(3, &BiscuitOS_counter);
	printk("[2]Atomic: %d\n", atomic_read(&BiscuitOS_counter));

	atomic_dec_unless_positive(&BiscuitOS_counter);

	/* New data */
	printk("[3]Atomic: %d\n", atomic_read(&BiscuitOS_counter));

	return 0;
}
device_initcall(atomic_demo_init);

#### <span id="驱动安装">驱动安装</span>

驱动的安装很简单，首先将驱动放到 drivers/BiscuitOS/ 目录下，命名为 atomic.c，
然后修改 Kconfig 文件，添加内容参考如下：

{% highlight bash %}
diff --git a/drivers/BiscuitOS/Kconfig b/drivers/BiscuitOS/Kconfig
index 4edc5a5..1a9abee 100644
--- a/drivers/BiscuitOS/Kconfig
+++ b/drivers/BiscuitOS/Kconfig
@@ -6,4 +6,14 @@ if BISCUITOS_DRV
config BISCUITOS_MISC
        bool "BiscuitOS misc driver"
+config BISCUITOS_ATOMIC
+       bool "atomic"
+
+if BISCUITOS_ATOMIC
+
+config DEBUG_BISCUITOS_ATOMIC
+       bool "atomic_dec_unless_positive"
+
+endif # BISCUITOS_ATOMIC
+
endif # BISCUITOS_DRV
{% endhighlight %}

接着修改 Makefile，请参考如下修改：

{% highlight bash %}
diff --git a/drivers/BiscuitOS/Makefile b/drivers/BiscuitOS/Makefile
index 82004c9..9909149 100644
--- a/drivers/BiscuitOS/Makefile
+++ b/drivers/BiscuitOS/Makefile
@@ -1 +1,2 @@
obj-$(CONFIG_BISCUITOS_MISC)     += BiscuitOS_drv.o
+obj-$(CONFIG_BISCUITOS_ATOMIC)  += atomic.o
--
{% endhighlight %}
