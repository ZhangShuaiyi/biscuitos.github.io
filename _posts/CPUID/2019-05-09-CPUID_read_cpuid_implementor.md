# <span id="源码分析">源码分析</span>

{% highlight ruby %}
static inline unsigned int __attribute_const__ read_cpuid_implementor(void)
{
        return (read_cpuid_id() & 0xFF000000) >> 24;
}
{% endhighlight %}

read_cpuid_implementor() 函数的主要功能就是获得 CPU 的生产商。标准定义中，
0x41 代表 ARM，0x69 代表 Intel，0x56 代表 Marvell 等，更多信息请参考

> [B5.1.105 MIDR,Main ID Register,VMSA](https://github.com/BiscuitOS/Documentation/blob/master/Datasheet/ARM/ARMv7_architecture_reference_manual.pdf)

#include <linux/kernel.h>
#include <linux/init.h>

#include <asm/cputype.h>

static __init int cpuid_demo_init(void)
{
	unsigned int type;

	type = read_cpuid_implementor();

	printk("CPU Implementor: %#x\n", type);

	return 0;
}
device_initcall(cpuid_demo_init);

#### <span id="驱动安装">驱动安装</span>

驱动的安装很简单，首先将驱动放到 drivers/BiscuitOS/ 目录下，命名为 cpuid.c，
然后修改 Kconfig 文件，添加内容参考如下：

{% highlight bash %}
diff --git a/drivers/BiscuitOS/Kconfig b/drivers/BiscuitOS/Kconfig
index 4edc5a5..1a9abee 100644
--- a/drivers/BiscuitOS/Kconfig
+++ b/drivers/BiscuitOS/Kconfig
@@ -6,4 +6,14 @@ if BISCUITOS_DRV
config BISCUITOS_MISC
        bool "BiscuitOS misc driver"
+config BISCUITOS_CPUID
+       bool "CPUID"
+
+if BISCUITOS_CPUID
+
+config DEBUG_BISCUITOS_CPUID
+       bool "read_cpuid_implementor"
+
+endif # BISCUITOS_CPUID
+
endif # BISCUITOS_DRV
{% endhighlight %}

接着修改 Makefile，请参考如下修改：

{% highlight bash %}
diff --git a/drivers/BiscuitOS/Makefile b/drivers/BiscuitOS/Makefile
index 82004c9..9909149 100644
--- a/drivers/BiscuitOS/Makefile
+++ b/drivers/BiscuitOS/Makefile
@@ -1 +1,2 @@
obj-$(CONFIG_BISCUITOS_MISC)     += BiscuitOS_drv.o
+obj-$(CONFIG_BISCUITOS_CPUID)  += cpuid.o
--
{% endhighlight %}
